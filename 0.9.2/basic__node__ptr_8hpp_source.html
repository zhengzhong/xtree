00169     //! Performs a static cast from pointer-to-U to pointer-to-T. Static cast is typically used to
00170     //! cast from subclass to superclass (up-cast), or to add constness to a node pointer.
00171     //!
00172     //! Typical use cases:
00173     //!
00174     //! \code
00175     //! // Add constness to pointer:
00176     //! basic_node_ptr<const element> ptr = static_node_cast<const element>(ptr_to_element);
00177     //! // Perform an up-cast:
00178     //! basic_node_ptr<node> ptr = static_node_cast<element>(ptr_to_element);
00179     //! \endcode
00180     //!
00181     //! \param ptr  the pointer-to-U to cast.
00182     //! \return the resulting pointer-to-T.
00183     template<class T, class U>
00184     inline basic_node_ptr<T> static_node_cast(const basic_node_ptr<U>& ptr)
00185     {
00186         return basic_node_ptr<T>( ptr, typename basic_node_ptr<T>::static_cast_tag() );
00187     }
