00288         {
00289             // Ensure the current position is not end.
00290             if (current_ == 0)
00291             {
00292                 throw bad_dom_operation("increment() called on end position");
00293             }
00294             // Try to move deeper.
00295             basic_node_ptr<Element> next = current_->find_first_elem();
00296             if (next != 0)
00297             {
00298                 current_ = next;
00299                 return;
00300             }
00301             // Try to move to next sibling element.
00302             next = current_->get_next_sibling_elem();
00303             if (next != 0)
00304             {
00305                 current_ = next;
00306                 return;
00307             }
00308             // Try to move to next sibling element of ancestors.
00309             while (current_ != parent_ && current_->parent() != parent_)
00310             {
00311                 current_ = current_->parent();
00312                 next = current_->get_next_sibling_elem();
00313                 if (next != 0)
00314                 {
00315                     current_ = next;
00316                     return;

00318             }

00324         {
00325             // Ensure the current position is neither begin nor end.
00326             if (current_ == 0)
