00070     //! \param qname  the QName to split.
00071     //! \return a pair of namespace prefix and local name.
00072     inline std::pair<std::string, std::string> split_qname(const xmlChar* qname)
00073     {
00074         std::string qname_str = to_chars(qname);
00075         std::string::size_type pos = qname_str.find_first_of(":");
00076         if (pos == std::string::npos)
00077         {
00078             return std::make_pair(std::string(), qname_str);
00079         }
00080         else
00081         {
00082             return std::make_pair(qname_str.substr(0, pos), qname_str.substr(pos + 1));
00083         }
00084     }
