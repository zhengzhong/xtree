00024     //! This class represents a phoenix singleton holder (see Modern C++ Design [MC++D], by Andrei
00025     //! Alexandrescu). A phoenix singleton class has the following states:
00026     //!
00027     //! - non-initialized: ( instance_ == 0 && destroyed_ == false )
00028     //!
00029     //!   This is the initial state guaranteed by the compiler. In this state, the singleton has
00030     //!   not been created, thus the compiler will never destroy the singleton on this state.
00031     //!
00032     //!   If instance() is called on this state, it will firstly delegate to create_singleton() to
00033     //!   create the singleton and change the state to initialized. Then it will return a reference
00034     //!   to the singleton instance.
00035     //!
00036     //! - initialized ( instance_ != 0 && destroyed_ == false )
00037     //!
00038     //!   In this state, the singleton has been created yet not been destroyed. The compiler may
00039     //!   decide to destroy the singleton at program exit, in an arbitrary order comparing to the
00040     //!   destructions of other singletons or static/global variables. The compiler destroy the
00041     //!   singleton by calling its destructor, which will change the state to destroyed.
00042     //!
00043     //!   If instance() is called on this state, it returns a reference to the singleton instance
00044     //!   directly.
00045     //!
00046     //! - destroyed: ( instance_ == 0 && destroyed_ == true )
00047     //!
00048     //!   In this state, the singleton has been destroyed by the compiler at program exit. Since
00049     //!   the singleton is declared as a static local variable in create_singleton(), although it
00050     //!   has already been destroyed, the compiler guarantee that static objects' memory lasts for
00051     //!   the duration of the program (MC++D, p120).
00052     //!
00053     //!   If instance() is called on this state, a dead reference is detected (someone is trying to
00054     //!   access the singleton instance after its destruction). It will firstly delegate to
00055     //!   rebear_singleton() to re-create the singleton instance in the shell of the old one,
00056     //!   register this new singleton's destruction with std::atexit(), and change the state to
00057     //!   initialized. Then it will return a reference to the re-born singleton instance.
00058     template<class T>
00059     class phoenix_singleton
00060     {
00061 
00062         typedef T                    singleton_type; //!< The underlying singleton type.
00063         typedef phoenix_singleton<T> this_type;      //!< Type of the templatized class.
00064 
00065     public:
00066 
00067         ////////////////////////////////////////////////////////////////////////////////////////////
00068         // public static singleton instance getter
00069         //
00070 
00071         //! Returns the singleton instance.
00072         //! \return the singleton instance.
00073         static singleton_type& instance()
00074         {
00075             if (instance_ == 0)
00076             {
00077                 if (!destroyed_)
00078                 {
00079                     // state is 'non-initialized': delegate to create_singleton().
00080                     create_singleton();
00081                 }
00082                 else
00083                 {
00084                     // state is 'destroyed': delegate to reborn_singleton().
00085                     rebear_singleton();
00086                 }
00087             }
00088             return *instance_;
00089         }
00090 
00091     protected:
00092 
00093         ////////////////////////////////////////////////////////////////////////////////////////////
00094         // protected constructor and destructor
00095         //
00096 
00097         //! Protected constructor to prevent external instantiation.
00098         explicit phoenix_singleton()
00099         {
00100             // Do nothing.
00101         }
00102 
00103         //! Protected constructor to prevent external destruction. This function sets the static
00104         //! pointer to the singleton instance to 0, and the static destroyed flag to true. In this
00105         //! way, the current state will change to destroyed after the destructor.
00106         virtual ~phoenix_singleton()
00107         {
00108             destroyed_ = true;
00109             instance_ = 0;
00110         }
00111 
00112     private:
00113 
00114         ////////////////////////////////////////////////////////////////////////////////////////////
00115         // static life-cycle management functions
00116         //
00117 
00118         //! Creates the (static) singleton object. This function may be called on two states:
00119         //! when the singleton is 'non-initialized'; or is 'destroyed'.
00120         //!
00121         //! In the first case, the static singleton variable will be created (this is the first
00122         //! time this function is invoked), and the destroyed flag was initialized to false by the
00123         //! compiler. Thus this function sets the static singleton pointer to the address of the
00124         //! newly-created static singleton variable.
00125         //!
00126         //! In the second case, this function is called by rebear_singleton(). The static singleton
00127         //! variable has already been created, and has been destroyed, and the destroyed flag was
00128         //! set to true (by the destructor). Thus this function sets the static singleton pointer
00129         //! to the original address (the shell) of the (destroyed) static singleton variable. It is
00130         //! then up to the rebear_singleton() to re-create the singleton in the shell.
00131         //!
00132         //! Note: although the static singleton object may be destroyed, its memory will last for
00133         //! the duration of the program (the property of static variables inside a function).
00134         static void create_singleton()
00135         {
00136             static singleton_type singleton;
00137             instance_ = &singleton;
00138         }
00139 
00140         //! This function is called when the current state is 'destroyed'. It re-creates the
00141         //! singleton in the shell of the old one, register the destruction of the new singleton
00142         //! object with std::atexit(), and change the state to 'initialized'.
00143         static void rebear_singleton()
00144         {
00145             create_singleton();               // obtain the shell of the destroyed singleton.
00146             new(instance_) singleton_type();  // placement new: re-create singleton in the shell.
00147             std::atexit(kill_singleton);      // register the destruction with std::atexit().
00148             destroyed_ = false;               // now the state is changed to 'initialized'.
00149         }
00150 
00151         //! If this function was registered with std::exit(), it will get called by the standard C
00152         //! library at program exit. It destroys the singleton (but does not free the memory --
00153         //! because the memory where the singleton is seated in was not allocated by the new
00154         //! operator), and change the state to 'destroyed'.
00155         static void kill_singleton()
00156         {
00157             // Call the destructor on the singleton instance pointer. Since the concrete singleton
00158             // class derives from this class template, the destructor of the phoenix singleton
00159             // class template will also get called, which will change the state to 'destroyed'.
00160             instance_->~singleton_type();
00161         }
00162 
00163     private:
00164 
00165         ////////////////////////////////////////////////////////////////////////////////////////////
00166         // noncopyable
00167         //
00168 
00169         //! Non-implemented copy constructor.
00170         phoenix_singleton(const this_type&);
00171 
00172         //! Non-implemented copy assignment.
00173         this_type& operator=(const this_type&);
00174 
00175     private:
00176 
00177         static singleton_type* instance_;  //!< The singleton instance.
00178         static bool            destroyed_; //!< Flag indicating if the singleton is destroyed.
00179 
00180     };
